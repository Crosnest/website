1. Install lz4

sudo apt-get update -y

sudo apt-get install liblz4-tool -y


2. Download & Decompress Quicksync Data after installing Crypto.org & chain-maind



FILENAME=crypto-org-chain-mainnet-1.20210601-2358.tar.lz4
cd ~/.chain-maind/
rm -rf data/ (if you already have data folder present)

wget https://quicksync.cros-nest.com/data/$FILENAME
wget https://quicksync.cros-nest.com/data/$FILENAME.checksum
curl -sSL https://quicksync.cros-nest.com/files/checksum.sh | bash -s -- $FILENAME
lz4 -d $FILENAME | tar xf -
